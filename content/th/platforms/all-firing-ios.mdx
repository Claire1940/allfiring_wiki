  </Step>
  <Step num="3" title="ควบคุมคอมโบและทักษะ">

  </Step>
  <Step num="4" title="การรับรู้สิ่งแวดล้อม">

  </Step>
  <Step num="5" title="เตรียมตัวและอัปเกรด">
